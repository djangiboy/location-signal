-- =============================================================================
-- DELHI ZONE • DEC 2025 • UNIFIED (INTEREST + ASSIGN + DECLINE) COHORT
-- One row per (mobile, partner_id). rn=1 on first decision (ASC).
-- DECLINED rows exclude source = 'System_Force_Declined_Post_Assigned_72hours'.
-- Install join unchanged: same partner installed the same mobile later.
-- =============================================================================

WITH
delhi_mobiles AS (
    SELECT DISTINCT mobile
    FROM booking_logs
    WHERE added_time >= '2025-12-01'
      AND added_time <  '2026-01-01'
      AND (
            LOWER(PARSE_JSON(data):city::STRING)  = 'delhi'
         OR LOWER(PARSE_JSON(data):zone::STRING)  = 'delhi'
         OR LOWER(PARSE_JSON(data):state::STRING) LIKE '%delhi%'
      )
),

allocation AS (
    SELECT mobile, partner_id, nearest_distance, nearest_lat, nearest_lng, created_at, nearest_type, probability
    FROM (
        SELECT
            mobile                                                                      AS mobile,
            ns.value:partner_id::STRING                                                 AS partner_id,
            ns.value:reason:nearest_type::STRING                                        AS nearest_type,
            TRY_CAST(NULLIF(ns.value:reason:nearest_distance::STRING, '') AS DOUBLE)    AS nearest_distance,
            TRY_CAST(NULLIF(SPLIT_PART(ns.value:reason:nearest_lat_lng::STRING,',',1), '') AS DOUBLE) AS nearest_lat,
            TRY_CAST(NULLIF(SPLIT_PART(ns.value:reason:nearest_lat_lng::STRING,',',2), '') AS DOUBLE) AS nearest_lng,
            TRY_CAST(NULLIF(ns.value:reason:probability::STRING, '') AS DOUBLE)         AS probability,
            created_at,
            ROW_NUMBER() OVER (
                PARTITION BY mobile, ns.value:partner_id::STRING
                ORDER BY created_at DESC
            ) AS rn,
            data
        FROM prod_db.mysql_rds_genie_genie2.t_allocation_logs,
        LATERAL FLATTEN(input => PARSE_JSON(data):notification_schedule) ns
        WHERE created_at >= '2025-12-01'
          AND created_at <  '2026-01-01'
    )
    WHERE rn = 1
),

decisions AS (
    SELECT mobile, partner_id, decision_time, event_name, decision_reason, decision_source
    FROM (
        SELECT
            mobile                                                  AS mobile,
            account_id::STRING                                      AS partner_id,
            TRY_PARSE_JSON(data):remarks::STRING                    AS decision_reason,
            TRY_PARSE_JSON(data):source::STRING                     AS decision_source,
            event_name                                              AS event_name,
            added_time                                              AS decision_time,
            ROW_NUMBER() OVER (
                PARTITION BY mobile, account_id::STRING
                ORDER BY added_time
            ) AS rn
        FROM prod_db.public.task_logs
        WHERE added_time >= '2025-12-01'
          AND added_time <  '2026-01-01'
          AND (
                event_name IN ('INTERESTED', 'ASSIGNED')
             OR (
                    event_name = 'DECLINED'
                AND (TRY_PARSE_JSON(data):source::STRING != 'System_Force_Declined_Post_Assigned_72hours'
                     OR TRY_PARSE_JSON(data):source::STRING IS NULL)
                )
          )
    ) a
    WHERE rn = 1
),

booking_location AS (
    SELECT mobile, booking_lat, booking_lng, g1_time
    FROM (
        SELECT
            mobile                                                                          AS mobile,
            TRY_CAST(NULLIF(SPLIT_PART(
                PARSE_JSON(response):genie_downstream_context:g1_lead_coords::STRING, ',', 1
            ), '') AS DOUBLE)                                                                  AS booking_lat,
            TRY_CAST(NULLIF(SPLIT_PART(
                PARSE_JSON(response):genie_downstream_context:g1_lead_coords::STRING, ',', 2
            ), '') AS DOUBLE)                                                                  AS booking_lng,
            created_at                                                                      AS g1_time,
            ROW_NUMBER() OVER (
                PARTITION BY mobile
                ORDER BY created_at DESC
            ) AS rn
        FROM mysql_rds_genie_genie1.t_serviceability_logs
        WHERE PARSE_JSON(response):serviceable::BOOLEAN = TRUE
          AND created_at >= '2025-11-20'
          AND created_at <  '2026-01-01'
    )
    WHERE rn = 1
),

booking_type AS (
    SELECT mobile
    FROM (
        SELECT
            mobile,
            ROW_NUMBER() OVER (PARTITION BY mobile ORDER BY added_time DESC) AS rn
        FROM booking_logs
        WHERE event_name = 'prospect_identified'
          AND added_time >= '2025-12-01'
          AND added_time <  '2026-01-01'
    ) a
    WHERE rn = 1
),

installed_decision AS (
    SELECT *
    FROM (
        SELECT
            mobile,
            added_time AS installed_time,
            account_id AS partner_id,
            ROW_NUMBER() OVER (PARTITION BY mobile ORDER BY added_time DESC) AS rn
        FROM prod_db.public.task_logs
        WHERE event_name = 'OTP_VERIFIED'
          AND added_time >= '2025-12-01'
          AND added_time <  '2026-01-31'
    ) a
    WHERE rn = 1
),

cohort AS (
    SELECT
        al.mobile,
        al.partner_id,
        al.nearest_distance,
        al.nearest_type,
        al.probability,
        al.nearest_lat,
        al.nearest_lng,
        bl.booking_lat,
        bl.booking_lng,
        bl.g1_time,
        al.created_at           AS allocated_at,
        d.decision_time,
        d.event_name            AS decision_event,
        d.decision_reason,
        d.decision_source,
        CASE WHEN bt.mobile IS NOT NULL THEN 1 ELSE 0 END AS bdo_lead,
        CASE WHEN id.partner_id = al.partner_id THEN 1 ELSE 0 END AS installed,
        id.installed_time
    FROM delhi_mobiles dm
    INNER JOIN allocation       al ON dm.mobile::STRING     = al.mobile::STRING
    INNER JOIN decisions        d  ON al.mobile::STRING     = d.mobile::STRING
                                  AND al.partner_id::STRING = d.partner_id::STRING
                                  AND DATEDIFF(seconds, al.created_at, d.decision_time) > 0
    INNER JOIN booking_location bl ON al.mobile::STRING     = bl.mobile::STRING
    LEFT  JOIN booking_type     bt ON al.mobile::STRING     = bt.mobile::STRING
    LEFT  JOIN installed_decision id ON al.mobile::STRING     = id.mobile::STRING
                                    AND al.partner_id::STRING = id.partner_id::STRING
                                    AND DATEDIFF(seconds, al.created_at, id.installed_time) >= 0
)
SELECT * FROM cohort;
